/**
 *
 *
 * ElementState is static class and provies common states of Element as Constants.
 */
class ElementState {


    /**
     *
     *
     * @readonly
     * @static
     * @memberof ElementState
     */
    static get LOADING() {
        return "loading";
    }


    /**
     *
     *
     * @readonly
     * @static
     * @memberof ElementState
     */
    static get LOADED() {
        return "loaded";
    }


    /**
     *
     *
     * @readonly
     * @static
     * @memberof ElementState
     */
    static get SUCCESS() {
        return "success";
    }


    /**
     *
     *
     * @readonly
     * @static
     * @memberof ElementState
     */
    static get ERROR() {
        return "error";
    }


    /**
     *
     *
     * @readonly
     * @static
     * @memberof ElementState
     */
    static get WARNING() {
        return "warning";
    }


    /**
     *
     *
     * @readonly
     * @static
     * @memberof ElementState
     */
    static get READONLY() {
        return "readonly";
    }


    /**
     *
     *
     * @readonly
     * @static
     * @memberof ElementState
     */
    static get ENABLED() {
        return "enabled";
    }


    /**
     *
     *
     * @readonly
     * @static
     * @memberof ElementState
     */
    static get DISABLED() {
        return "disabled";
    }


    /**
     *
     *
     * @readonly
     * @static
     * @memberof ElementState
     */
    static get DEFAULT() {
        return "default";
    }

}

export default ElementState;
